file(GLOB QUEUE_TEST_SOURCES "*.cpp")

foreach(test_file ${QUEUE_TEST_SOURCES})
    get_filename_component(test_name ${test_file} NAME_WE)
    set(test_target "test_queue_${test_name}")
    add_executable(${test_target} ${test_file})
    target_link_libraries(${test_target} PRIVATE common_queue)
    add_test(NAME ${test_target} COMMAND ${test_target})
    message(STATUS "Added queue test: ${test_target}")
endforeach()
